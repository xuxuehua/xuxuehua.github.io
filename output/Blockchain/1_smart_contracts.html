Smart Contracts

Hello World 代码

pragma solidity ^0.4.21; // 程序版本

contract SimpleStorage {       // 合约声明
    uint storedData;

    function set(uint x) {
        storedData = x;
    }

    function get() constant returns (uint){
        return storedData;
    }
}

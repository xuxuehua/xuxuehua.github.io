import 模块合并

const atImport = require('postcss-import');

module.exports = {
        plugins: [
                atImport,
                autoprefixer({
                        browser:['Firefox > 30']
                }),
        ]
};

autoprefixer自动加前缀

const autoprefixer = require('autoprefixer');

module.exports = {
        plugins: [
                autoprefixer({
                        browsers:['Firefox > 30']
                }),
        ]
};

postcss src/plugins-main.css -o build/plugins-main.css

cssnano 压缩代码

const autoprefixer = require('autoprefixer');

module.exports = {
        plugins: [
                autoprefixer({
                        browsers:['Firefox > 30']
                }),
                cssnano
        ]
};

postcss src/plugins-main.css -o build/plugins-main.css

.bash_profile

function proxy_off(){
    unset http_proxy
    unset https_proxy
    echo -e "proxy_off"
}

function proxy_on() {
    /usr/local/sbin/privoxy /usr/local/etc/privoxy/config
    export no_proxy="localhost,127.0.0.1,localaddress,.localdomain.com"
    export http_proxy="http://127.0.0.1:8118"
    export https_proxy=$http_proxy
    echo -e "proxy_on"
}
